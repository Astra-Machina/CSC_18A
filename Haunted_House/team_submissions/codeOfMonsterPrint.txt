        public void skeletonPrintYoSelf(){

            output.append("   .-\"\"-. \n")
                  .append("  / _  _ \\\n")
                  .append("  |(_)(_)|\n")
                  .append("  (_ /\\ _)\n")
                  .append("   |wwww| \n")

        }
